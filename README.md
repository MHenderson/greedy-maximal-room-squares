# more-maximal-room-squares

<!-- badges: start -->
<!-- badges: end -->

``` r
n <- 10

r <- Room$new(size = n)

tic()
for(e in r$empty_cells) {
  for(p in r$free_pairs) {
    if(r$is_available(e, p)) {
      r$set(e, p)
      break()
    }
  }
}
toc()
#> 0.314 sec elapsed
```
